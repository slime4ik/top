from django import forms
from django.core.exceptions import ValidationError
from django.contrib.auth.models import User
import os

class TrashForm(forms.Form):
    description = forms.CharField(
        max_length=100,
        required=False,
        widget=forms.Textarea(attrs={'placeholder': 'Описание изображение'})
    )
    image = forms.ImageField()
    
    def clean_image(self):
        image = self.cleaned_data.get('image')

        #Проверяем формат файла
        valid_extensions = ['jpg', 'jpeg', 'png', 'gif']
        ext = os.path.splitext(image.name)[1][1:].lower()
        if ext not in valid_extensions:
            raise ValidationError('Формат файла не поддерживается. Разрешены только jpg, jpeg, png, gif')
        
        #Проверяем размер файла
        max_size = 2 * 1024 * 1024  #2 мб
        if image.size > max_size:
            raise ValidationError('размер файла не должен превышать 2 мб')
        
        #Проверяем нет ли в названии файла пробелов
        if ' ' in image.name:
            raise ValidationError('в названии файла не должно быть пробелов')
        return image
